import { getAuth, onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-auth.js"; const auth = getAuth(); // The "Guard" Listener onAuthStateChanged(auth, (user) => { if (user) { // 1. User is signed in console.log("Access Granted:", user.email); document.getElementById('user-display').textContent = user.email; // You can now safely fetch private clan data here } else { // 2. No user is signed in console.log("Access Denied. Redirecting..."); window.location.href = "login.html"; // Kick unauthorized users back to login } }); // Logout Utility document.getElementById('logout-btn').addEventListener('click', () => { signOut(auth).then(() => { window.location.href = "login.html"; }); });
